-- ============================================
-- COMPLETE STORAGE SETUP WITH POLICIES
-- Jalankan ini jika script pertama masih error RLS
-- ============================================

-- Step 1: Cleanup (hapus bucket lama)
DELETE FROM storage.objects WHERE bucket_id = 'berita-thumbnails';
DELETE FROM storage.buckets WHERE id = 'berita-thumbnails';

-- Step 2: Create bucket
INSERT INTO storage.buckets (
  id, 
  name, 
  public, 
  file_size_limit, 
  allowed_mime_types,
  avif_autodetection
)
VALUES (
  'berita-thumbnails',
  'berita-thumbnails',
  true,
  5242880,
  ARRAY['image/jpeg', 'image/jpg', 'image/png', 'image/webp', 'image/x-webp'],
  false
);

-- Step 3: Create policies (untuk semua roles)

-- Policy 1: Public SELECT (anyone can view)
CREATE POLICY "public_select_berita_thumbnails"
ON storage.objects FOR SELECT
TO public
USING (bucket_id = 'berita-thumbnails');

-- Policy 2: Authenticated INSERT
CREATE POLICY "authenticated_insert_berita_thumbnails"
ON storage.objects FOR INSERT
TO authenticated
WITH CHECK (bucket_id = 'berita-thumbnails');

-- Policy 3: Authenticated UPDATE
CREATE POLICY "authenticated_update_berita_thumbnails"
ON storage.objects FOR UPDATE
TO authenticated
USING (bucket_id = 'berita-thumbnails')
WITH CHECK (bucket_id = 'berita-thumbnails');

-- Policy 4: Authenticated DELETE
CREATE POLICY "authenticated_delete_berita_thumbnails"
ON storage.objects FOR DELETE
TO authenticated
USING (bucket_id = 'berita-thumbnails');

-- Policy 5: Anon INSERT (untuk upload tanpa auth)
CREATE POLICY "anon_insert_berita_thumbnails"
ON storage.objects FOR INSERT
TO anon
WITH CHECK (bucket_id = 'berita-thumbnails');

-- Policy 6: Anon UPDATE
CREATE POLICY "anon_update_berita_thumbnails"
ON storage.objects FOR UPDATE
TO anon
USING (bucket_id = 'berita-thumbnails')
WITH CHECK (bucket_id = 'berita-thumbnails');

-- Policy 7: Anon DELETE
CREATE POLICY "anon_delete_berita_thumbnails"
ON storage.objects FOR DELETE
TO anon
USING (bucket_id = 'berita-thumbnails');

-- Step 4: Verify setup
SELECT 
  'Bucket created:' as status,
  id, 
  name, 
  public as is_public,
  file_size_limit / 1024 / 1024 as size_limit_mb
FROM storage.buckets 
WHERE id = 'berita-thumbnails'

UNION ALL

SELECT 
  'Policies created:' as status,
  COUNT(*)::text as id,
  'policies total' as name,
  NULL as is_public,
  NULL as size_limit_mb
FROM pg_policies 
WHERE schemaname = 'storage' 
  AND tablename = 'objects'
  AND policyname LIKE '%berita_thumbnails%';

-- Lihat detail policies
SELECT 
  policyname as policy_name,
  cmd as operation,
  roles as target_roles
FROM pg_policies 
WHERE schemaname = 'storage' 
  AND tablename = 'objects'
  AND policyname LIKE '%berita_thumbnails%'
ORDER BY policyname;

-- 1. Drop trigger lama
DROP TRIGGER IF EXISTS cleanup_old_thumbnail ON berita;
DROP TRIGGER IF EXISTS cleanup_thumbnail_on_delete ON berita;

-- 2. Drop function lama jika ada
DROP FUNCTION IF EXISTS delete_old_thumbnail();
DROP FUNCTION IF EXISTS delete_thumbnail_on_delete();

-- 3. Buat function baru yang benar
CREATE OR REPLACE FUNCTION delete_old_thumbnail()
RETURNS TRIGGER AS $$
BEGIN
  -- Hanya jalankan jika thumbnail berubah
  IF OLD.thumbnail IS NOT NULL AND OLD.thumbnail IS DISTINCT FROM NEW.thumbnail THEN
    -- Extract path dari URL
    DECLARE
      old_path TEXT;
    BEGIN
      -- Format URL: https://xxx.supabase.co/storage/v1/object/public/berita-thumbnails/path/to/file.jpg
      old_path := regexp_replace(OLD.thumbnail, '^.*/berita-thumbnails/', '');
      
      -- Hapus file dari storage
      PERFORM extensions.http((
        'DELETE',
        current_setting('app.settings.supabase_url') || '/storage/v1/object/berita-thumbnails/' || old_path,
        ARRAY[extensions.http_header('apikey', current_setting('app.settings.supabase_anon_key'))],
        NULL,
        NULL
      )::extensions.http_request);
    EXCEPTION WHEN OTHERS THEN
      -- Ignore error jika gagal hapus file
      NULL;
    END;
  END IF;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 4. Buat function untuk delete
CREATE OR REPLACE FUNCTION delete_thumbnail_on_delete()
RETURNS TRIGGER AS $$
BEGIN
  IF OLD.thumbnail IS NOT NULL THEN
    DECLARE
      old_path TEXT;
    BEGIN
      old_path := regexp_replace(OLD.thumbnail, '^.*/berita-thumbnails/', '');
      
      PERFORM extensions.http((
        'DELETE',
        current_setting('app.settings.supabase_url') || '/storage/v1/object/berita-thumbnails/' || old_path,
        ARRAY[extensions.http_header('apikey', current_setting('app.settings.supabase_anon_key'))],
        NULL,
        NULL
      )::extensions.http_request);
    EXCEPTION WHEN OTHERS THEN
      NULL;
    END;
  END IF;
  
  RETURN OLD;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
